"#version 330\n"
"in vec2 fragTexCoord;\n"
"in vec4 fragColor;\n"
"out vec4 finalColor;\n"
"uniform sampler2D texture0;\n"
"uniform float threshold;\n"
"uniform float intensity;\n"
"uniform float radius;\n"
"uniform vec2 resolution;\n"
"void main() {\n"
"    vec4 color = texture(texture0, fragTexCoord);\n"
"    vec2 ts = vec2(radius) / resolution;\n"
"    vec4 bloom = vec4(0.0);\n"
"    bloom += max(texture(texture0, fragTexCoord + vec2(-ts.x, -ts.y)) - vec4(threshold), vec4(0.0));\n"
"    bloom += max(texture(texture0, fragTexCoord + vec2( 0.0,  -ts.y)) - vec4(threshold), vec4(0.0));\n"
"    bloom += max(texture(texture0, fragTexCoord + vec2( ts.x, -ts.y)) - vec4(threshold), vec4(0.0));\n"
"    bloom += max(texture(texture0, fragTexCoord + vec2(-ts.x,  0.0 )) - vec4(threshold), vec4(0.0));\n"
"    bloom += max(texture(texture0, fragTexCoord + vec2( ts.x,  0.0 )) - vec4(threshold), vec4(0.0));\n"
"    bloom += max(texture(texture0, fragTexCoord + vec2(-ts.x,  ts.y)) - vec4(threshold), vec4(0.0));\n"
"    bloom += max(texture(texture0, fragTexCoord + vec2( 0.0,   ts.y)) - vec4(threshold), vec4(0.0));\n"
"    bloom += max(texture(texture0, fragTexCoord + vec2( ts.x,  ts.y)) - vec4(threshold), vec4(0.0));\n"
"    bloom /= 8.0;\n"
"    finalColor = clamp(color + bloom * intensity, 0.0, 1.0) * fragColor;\n"
"}"
