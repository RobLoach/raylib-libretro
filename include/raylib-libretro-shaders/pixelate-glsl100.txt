"#version 100\n"
"precision mediump float;\n"
"varying vec2 fragTexCoord;\n"
"varying vec4 fragColor;\n"
"uniform sampler2D texture0;\n"
"uniform float pixelWidth;\n"
"uniform float pixelHeight;\n"
"uniform vec2 resolution;\n"
"void main() {\n"
"    vec2 pixSize = vec2(pixelWidth / resolution.x, pixelHeight / resolution.y);\n"
"    vec2 coord = floor(fragTexCoord / pixSize) * pixSize + pixSize * 0.5;\n"
"    gl_FragColor = texture2D(texture0, coord) * fragColor;\n"
"}"
