"#version 100\n"
"precision mediump float;\n"
"varying vec2 fragTexCoord;\n"
"varying vec4 fragColor;\n"
"uniform sampler2D texture0;\n"
"uniform float radius;\n"
"uniform float softness;\n"
"uniform float opacity;\n"
"uniform vec3 tintColor;\n"
"void main() {\n"
"    vec4 texColor = texture2D(texture0, fragTexCoord);\n"
"    float dist = length(fragTexCoord - vec2(0.5));\n"
"    float vignette = smoothstep(radius, radius - softness, dist);\n"
"    vec3 vigColor = mix(tintColor, vec3(1.0), vignette);\n"
"    gl_FragColor = vec4(texColor.rgb * mix(vec3(1.0), vigColor, opacity), texColor.a) * fragColor;\n"
"}"
