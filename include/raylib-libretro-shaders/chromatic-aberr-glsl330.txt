"#version 330\n"
"in vec2 fragTexCoord;\n"
"in vec4 fragColor;\n"
"out vec4 finalColor;\n"
"uniform sampler2D texture0;\n"
"uniform float redOffset;\n"
"uniform float greenOffset;\n"
"uniform float blueOffset;\n"
"uniform float strength;\n"
"void main() {\n"
"    vec2 dir = fragTexCoord - vec2(0.5);\n"
"    float r = texture(texture0, fragTexCoord + dir * redOffset   * strength).r;\n"
"    float g = texture(texture0, fragTexCoord + dir * greenOffset * strength).g;\n"
"    float b = texture(texture0, fragTexCoord + dir * blueOffset  * strength).b;\n"
"    float a = texture(texture0, fragTexCoord).a;\n"
"    finalColor = vec4(r, g, b, a) * fragColor;\n"
"}"
