"#version 120\n"
"varying vec2 fragTexCoord;\n"
"varying vec4 fragColor;\n"
"uniform sampler2D texture0;\n"
"uniform float redOffset;\n"
"uniform float greenOffset;\n"
"uniform float blueOffset;\n"
"uniform float strength;\n"
"void main() {\n"
"    vec2 dir = fragTexCoord - vec2(0.5);\n"
"    float r = texture2D(texture0, fragTexCoord + dir * redOffset   * strength).r;\n"
"    float g = texture2D(texture0, fragTexCoord + dir * greenOffset * strength).g;\n"
"    float b = texture2D(texture0, fragTexCoord + dir * blueOffset  * strength).b;\n"
"    float a = texture2D(texture0, fragTexCoord).a;\n"
"    gl_FragColor = vec4(r, g, b, a) * fragColor;\n"
"}"
